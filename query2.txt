function printresultidprice($result) { //prints results from a select statement
  echo "<table>";
  echo "<tr><th>TicketID</th><th>TicketPrice</th></tr>";

  while ($row = OCI_Fetch_Array($result, OCI_BOTH)) {
    echo "<tr><td>" . $row[0] . "</td><td>" . $row[1] ."</td></tr>" ;
}
echo "</table>";    
}
